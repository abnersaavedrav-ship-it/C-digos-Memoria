import numpy as np
from typing import Tuple, List, Dict
from dataclasses import dataclass

# ============================================================
# BLOQUE 0 — Utilidades base (robustas para NumPy >= 2.0)
# ============================================================
def _asarray_f64(a) -> np.ndarray:
    return np.asarray(a, dtype=np.float64)

def _nanmean(a: np.ndarray) -> float:
    a = _asarray_f64(a)
    m = np.isfinite(a)
    return float(a[m].mean()) if m.any() else np.nan

def _nanvar(a: np.ndarray, ddof: int = 0) -> float:
    a = _asarray_f64(a)
    m = np.isfinite(a)
    n = int(m.sum())
    if n == 0 or n - ddof <= 0:
        return np.nan
    mu = float(a[m].mean())
    return float(((a[m] - mu)**2).sum() / (n - ddof))

def _nanstd(a: np.ndarray, ddof: int = 0) -> float:
    v = _nanvar(a, ddof=ddof)
    return float(np.sqrt(v)) if np.isfinite(v) else np.nan

def tukey_window_2d(n1: int, n2: int, alpha: float = 0.5) -> np.ndarray:
    def tukey_1d(n, a):
        n = int(n)
        if a <= 0:  return np.ones(n, dtype=np.float64)
        if a >= 1:  return np.hanning(n).astype(np.float64)
        w = np.ones(n, dtype=np.float64)
        edge = int(a*(n-1)/2.0)
        if edge > 0:
            x = np.linspace(0, np.pi, edge, endpoint=False)
            w[:edge] = 0.5*(1 - np.cos(x))
            w[-edge:] = 0.5*(1 - np.cos(x[::-1]))
        return w
    return np.outer(tukey_1d(n1, alpha), tukey_1d(n2, alpha))

def zscore_pairwise(X: np.ndarray, Y: np.ndarray, eps: float = 1e-12) -> Tuple[np.ndarray, np.ndarray]:
    X = _asarray_f64(X); Y = _asarray_f64(Y)
    Xc = X - _nanmean(X);  Yc = Y - _nanmean(Y)
    Xs = Xc / (max(_nanstd(Xc), 0.0) + eps)
    Ys = Yc / (max(_nanstd(Yc), 0.0) + eps)
    return Xs, Ys

# ============================================================
# BLOQUE 1 — Matérn 2D espectral con anisotropía + LMC bivariante
# ============================================================
@dataclass
class MaternParams:
    nu: float
    ell: float
    sill: float
    anis_ratio: float
    angle_deg: float

def _freq_grids(n1: int, n2: int) -> Tuple[np.ndarray, np.ndarray]:
    k1 = np.fft.fftfreq(n1) * (2*np.pi)
    k2 = np.fft.fftfreq(n2) * (2*np.pi)
    K1, K2 = np.meshgrid(k2, k1)  # (fila, columna) -> (y,x)
    return K1, K2

def _anisotropic_quadform(K1: np.ndarray, K2: np.ndarray, anis_ratio: float, angle_deg: float) -> np.ndarray:
    a = float(max(anis_ratio, 1.0))
    theta = np.deg2rad(angle_deg)
    c, s = np.cos(theta), np.sin(theta)
    u = c*K1 + s*K2
    v = -s*K1 + c*K2
    u_scaled = u / a
    v_scaled = v * a
    return u_scaled**2 + v_scaled**2

def _matern_spectral_density(K1: np.ndarray, K2: np.ndarray, nu: float, ell: float,
                             anis_ratio: float, angle_deg: float) -> np.ndarray:
    kappa2 = (2.0*nu) / (ell**2)
    quad = _anisotropic_quadform(K1, K2, anis_ratio, angle_deg)
    expo = -(nu + 1.0)  # d/2=1 en 2D
    S = (kappa2 + quad) ** expo
    return np.maximum(S, 0.0)

def simulate_matern_fft(n1: int, n2: int, params: MaternParams, rng: np.random.Generator) -> np.ndarray:
    K1, K2 = _freq_grids(n1, n2)
    S = _matern_spectral_density(K1, K2, params.nu, params.ell, params.anis_ratio, params.angle_deg)
    Z_real = rng.normal(size=(n1, n2))
    Z_imag = rng.normal(size=(n1, n2))
    W = Z_real + 1j*Z_imag
    F = np.sqrt(S) * W
    X = np.fft.ifft2(F).real
    X = X / (np.std(X) + 1e-12) * np.sqrt(params.sill)
    return X

def lmc_pair_matern(n1: int, n2: int, p1: MaternParams, p2: MaternParams,
                    rho0: float, rng: np.random.Generator) -> Tuple[np.ndarray, np.ndarray]:
    Z1 = simulate_matern_fft(n1, n2, MaternParams(p1.nu, p1.ell, 1.0, p1.anis_ratio, p1.angle_deg), rng)
    Z2 = simulate_matern_fft(n1, n2, MaternParams(p2.nu, p2.ell, 1.0, p2.anis_ratio, p2.angle_deg), rng)
    Z1 = Z1 / (np.std(Z1) + 1e-12)
    Z2 = Z2 / (np.std(Z2) + 1e-12)
    X = Z1 * np.sqrt(p1.sill)
    Y = (rho0 * Z1 + np.sqrt(max(1.0 - rho0**2, 0.0)) * Z2) * np.sqrt(p2.sill)
    return X, Y

# ============================================================
# BLOQUE 2 — Codispersión (promedio de aportes locales)
# ============================================================
def codisp_rho_hat(X: np.ndarray, Y: np.ndarray, h: Tuple[int,int], mode: str = "toroidal") -> float:
    X = _asarray_f64(X); Y = _asarray_f64(Y)
    n1, n2 = X.shape
    h1, h2 = h
    if mode == "toroidal":
        I = (np.arange(n1)[:, None] + h1) % n1
        J = (np.arange(n2)[None, :] + h2) % n2
        dX = X[I, J] - X
        dY = Y[I, J] - Y
    else:
        I = np.arange(0, n1 - abs(h1)) if h1 >= 0 else np.arange(abs(h1), n1)
        J = np.arange(0, n2 - abs(h2)) if h2 >= 0 else np.arange(abs(h2), n2)
        if I.size == 0 or J.size == 0: return 0.0
        X0 = X[np.ix_(I, J)]; Y0 = Y[np.ix_(I, J)]
        I2 = I + (h1 if h1 >= 0 else -abs(h1))
        J2 = J + (h2 if h2 >= 0 else -abs(h2))
        dX = X[np.ix_(I2, J2)] - X0
        dY = Y[np.ix_(I2, J2)] - Y0
    Abar = float((dX**2).mean()); Bbar = float((dY**2).mean())
    denom = np.sqrt(max(Abar, 0.0) * max(Bbar, 0.0))
    if denom == 0.0: return 0.0
    S = (dX * dY) / denom
    return float(S.mean())

# ============================================================
# BLOQUE 3 — Remuestreos: BB (Lahiri) y CBB toroidal pareado
# ============================================================
def _list_overlapping_origins(n1: int, n2: int, b: int) -> List[Tuple[int,int]]:
    I = range(0, n1 - b + 1); J = range(0, n2 - b + 1)
    return [(i, j) for i in I for j in J]

def bb_lahiri_sample_pair(X: np.ndarray, Y: np.ndarray, b: int, rng: np.random.Generator) -> Tuple[np.ndarray,np.ndarray]:
    X = _asarray_f64(X); Y = _asarray_f64(Y)
    n1, n2 = X.shape
    M1 = (n1 // b) * b; M2 = (n2 // b) * b
    nB1 = M1 // b; nB2 = M2 // b
    Xs = np.zeros((M1, M2), dtype=np.float64)
    Ys = np.zeros((M1, M2), dtype=np.float64)
    origins = _list_overlapping_origins(n1, n2, b)
    n_orig = len(origins)
    for bi in range(nB1):
        for bj in range(nB2):
            oi, oj = origins[rng.integers(0, n_orig)]
            Xs[bi*b:(bi+1)*b, bj*b:(bj+1)*b] = X[oi:oi+b, oj:oj+b]
            Ys[bi*b:(bi+1)*b, bj*b:(bj+1)*b] = Y[oi:oi+b, oj:oj+b]
    return Xs, Ys

def cbb_sample_pair(X: np.ndarray, Y: np.ndarray, b: int, rng: np.random.Generator) -> Tuple[np.ndarray,np.ndarray]:
    X = _asarray_f64(X); Y = _asarray_f64(Y)
    n1, n2 = X.shape
    nB1 = int(np.ceil(n1 / b)); nB2 = int(np.ceil(n2 / b))
    Xs = np.zeros((nB1*b, nB2*b), dtype=np.float64)
    Ys = np.zeros_like(Xs)
    for bi in range(nB1):
        for bj in range(nB2):
            oi = rng.integers(0, n1)
            oj = rng.integers(0, n2)
            I = (oi + np.arange(b)) % n1
            J = (oj + np.arange(b)) % n2
            Xs[bi*b:(bi+1)*b, bj*b:(bj+1)*b] = X[np.ix_(I, J)]
            Ys[bi*b:(bi+1)*b, bj*b:(bj+1)*b] = Y[np.ix_(I, J)]
    return Xs[:n1, :n2], Ys[:n1, :n2]

def bootstrap_codisp(X: np.ndarray, Y: np.ndarray, H: List[Tuple[int,int]],
                     b: int, B: int, rng: np.random.Generator,
                     sampler: str = "BB", mode_rho: str = "toroidal"
) -> Dict[Tuple[int,int], Dict[str, np.ndarray]]:
    rho_star = {h: np.empty(B, dtype=np.float64) for h in H}
    for k in range(B):
        if sampler == "BB":
            Xs, Ys = bb_lahiri_sample_pair(X, Y, b, rng)
        elif sampler == "CBB":
            Xs, Ys = cbb_sample_pair(X, Y, b, rng)
        else:
            raise ValueError("sampler must be 'BB' or 'CBB'")
        for h in H:
            rho_star[h][k] = codisp_rho_hat(Xs, Ys, h, mode=mode_rho)
    out = {}
    for h in H:
        v = float(np.var(rho_star[h], ddof=1))
        lo = float(np.percentile(rho_star[h], 2.5))
        hi = float(np.percentile(rho_star[h], 97.5))
        out[h] = {"samples": rho_star[h], "var": v, "ci": (lo, hi)}
    return out

# BLOQUE 4 — Loop Monte Carlo SIN segunda pasada (IC guardados)
# ============================================================
@dataclass
class Scenario:
    rho0: float
    range_pix: float
    anis_mode: str      # 'iso' o 'aniso'
    anis_ratio: float
    angle_deg: float
    nu: float
    n1: int
    n2: int
    B_boot: int
    b_block: int
    R_mc: int

def run_mc_study_lmc(
    sc: Scenario,
    H: List[Tuple[int,int]],
    use_taper: bool = True,
    tukey_alpha: float = 0.5,
    seed: int = 12345
) -> Dict[Tuple[int,int], Dict[str, float]]:
    """
    ÚNICA pasada: guarda Var* e IC por réplica.
    Cobertura se calcula al final contra la media plug-in rho_mean_MC.
    """
    rng = np.random.default_rng(seed)

    # Parámetros Matérn (mismo nu y ell para simplicidad)
    p1 = MaternParams(nu=sc.nu, ell=sc.range_pix, sill=1.0,
                      anis_ratio=(sc.anis_ratio if sc.anis_mode=='aniso' else 1.0),
                      angle_deg=(sc.angle_deg if sc.anis_mode=='aniso' else 0.0))
    p2 = MaternParams(nu=sc.nu, ell=sc.range_pix, sill=1.0,
                      anis_ratio=(sc.anis_ratio if sc.anis_mode=='aniso' else 1.0),
                      angle_deg=(sc.angle_deg if sc.anis_mode=='aniso' else 0.0))

    R = sc.R_mc; B = sc.B_boot; b = sc.b_block
    n1, n2 = sc.n1, sc.n2

    # Buffers
    rho_hat = {h: np.empty(R, dtype=np.float64) for h in H}
    var_star_BB = {h: np.empty(R, dtype=np.float64) for h in H}
    var_star_CBB = {h: np.empty(R, dtype=np.float64) for h in H}
    ci_lo_BB = {h: np.empty(R, dtype=np.float64) for h in H}
    ci_hi_BB = {h: np.empty(R, dtype=np.float64) for h in H}
    ci_lo_CBB = {h: np.empty(R, dtype=np.float64) for h in H}
    ci_hi_CBB = {h: np.empty(R, dtype=np.float64) for h in H}

    # Réplicas MC (una sola pasada)
    for r in range(R):
        X, Y = lmc_pair_matern(n1, n2, p1, p2, sc.rho0, rng)
        X, Y = zscore_pairwise(X, Y)
        if use_taper:
            W = tukey_window_2d(n1, n2, alpha=tukey_alpha)
            X *= W; Y *= W

        # rho_hat original
        for h in H:
            rho_hat[h][r] = codisp_rho_hat(X, Y, h, mode="toroidal")

        # Bootstraps (única vez)
        out_BB  = bootstrap_codisp(X, Y, H, b=b, B=B, rng=rng, sampler="BB",  mode_rho="toroidal")
        out_CBB = bootstrap_codisp(X, Y, H, b=b, B=B, rng=rng, sampler="CBB", mode_rho="toroidal")

        # Guardar Var* e IC por réplica (para cobertura posterior)
        for h in H:
            var_star_BB[h][r]  = out_BB[h]["var"]
            var_star_CBB[h][r] = out_CBB[h]["var"]
            loB, hiB = out_BB[h]["ci"]
            loC, hiC = out_CBB[h]["ci"]
            ci_lo_BB[h][r], ci_hi_BB[h][r] = loB, hiB
            ci_lo_CBB[h][r], ci_hi_CBB[h][r] = loC, hiC

    # Media plug-in por lag
    rho_mean_MC = {h: float(rho_hat[h].mean()) for h in H}

    # Cobertura: compara rho_mean_MC con los IC guardados (sin re-bootstrap)
    cover_BB = {h: float(np.mean((ci_lo_BB[h] <= rho_mean_MC[h]) & (rho_mean_MC[h] <= ci_hi_BB[h])))
                for h in H}
    cover_CBB = {h: float(np.mean((ci_lo_CBB[h] <= rho_mean_MC[h]) & (rho_mean_MC[h] <= ci_hi_CBB[h])))
                 for h in H}

    # Agregados finales
    results = {}
    for h in H:
        var_mc = float(np.var(rho_hat[h], ddof=1))
        vbb = float(var_star_BB[h].mean())
        vcb = float(var_star_CBB[h].mean())
        results[h] = {
            "rho_hat_mean": rho_mean_MC[h],
            "Var_MC": var_mc,
            "Var*_BB_mean": vbb,
            "Var*_CBB_mean": vcb,
            "ratio_BB": (vbb / var_mc) if var_mc > 0 else np.nan,
            "ratio_CBB": (vcb / var_mc) if var_mc > 0 else np.nan,
            "coverage_BB": cover_BB[h],
            "coverage_CBB": cover_CBB[h],
        }
    return results

# BLOQUE 5 — Utilidades de impresión y GRID de escenarios
# ============================================================
def print_results_table(res_dict: Dict[Tuple[int,int], Dict[str, float]],
                        H: List[Tuple[int,int]],
                        header_label: str):
    header = f"[BB vs CBB] {header_label}"
    print("\n" + header)
    print("-" * len(header))
    subh = "h         | rho_mean   Var_MC      Var*_BB   Var*_CBB   ratio_BB  ratio_CBB  cov_BB  cov_CBB"
    print(subh)
    print("-"*len(subh))
    ratios_bb, ratios_cbb, cov_bb, cov_cbb = [], [], [], []
    rho_abs = []
    for h in H:
        r = res_dict[h]
        h_str = f"{h}"
        print(f"{h_str:<9} | {r['rho_hat_mean']:>+8.4f}  {r['Var_MC']:>8.2e}  "
              f"{r['Var*_BB_mean']:>8.2e}  {r['Var*_CBB_mean']:>9.2e}  "
              f"{r['ratio_BB']:>8.3f}  {r['ratio_CBB']:>9.3f}  "
              f"{r['coverage_BB']:>6.2f}  {r['coverage_CBB']:>7.2f}")
        if np.isfinite(r['ratio_BB']):  ratios_bb.append(r['ratio_BB'])
        if np.isfinite(r['ratio_CBB']): ratios_cbb.append(r['ratio_CBB'])
        cov_bb.append(r['coverage_BB']); cov_cbb.append(r['coverage_CBB'])
        rho_abs.append((h, abs(r['rho_hat_mean'])))
    mean_ratio_bb  = float(np.mean(ratios_bb))  if ratios_bb  else np.nan
    mean_ratio_cbb = float(np.mean(ratios_cbb)) if ratios_cbb else np.nan
    mean_cov_bb    = float(np.mean(cov_bb))     if cov_bb     else np.nan
    mean_cov_cbb   = float(np.mean(cov_cbb))    if cov_cbb    else np.nan
    h_star, rho_star = max(rho_abs, key=lambda t: t[1])
    print("\nResumen agregado (promedio sobre lags H):")
    print(f"  • ratio_BB medio   = {mean_ratio_bb:>6.3f}   (≈1 ideal)")
    print(f"  • ratio_CBB medio  = {mean_ratio_cbb:>6.3f}   (≈1 ideal)")
    print(f"  • cobertura_BB     = {mean_cov_bb:>6.3f}   (≈0.95 ideal)")
    print(f"  • cobertura_CBB    = {mean_cov_cbb:>6.3f}   (≈0.95 ideal)")
    print(f"  • h* (máx |rho|)   = {h_star}, |rho_mean| = {rho_star:.4f}")

def run_grid_study(
    n1: int, n2: int,
    nu: float,
    range_pix_list: List[float],
    rho0_list: List[float],
    anis_mode_list: List[str],
    b_list: List[int],
    H: List[Tuple[int,int]],
    B_boot: int = 800,
    R_mc: int = 100,
    anis_ratio: float = 2.0,
    angle_deg: float = 45.0,
    use_taper: bool = True,
    tukey_alpha: float = 0.5,
    seed_base: int = 2025
) -> Dict[Tuple[float, float, str, int], Dict[Tuple[int,int], Dict[str, float]]]:
    """
    Devuelve un diccionario indexado por (range_pix, rho0, anis_mode, b),
    cuyo valor es el 'results' por-lag de run_mc_study_lmc.
    """
    all_out = {}
    # total de escenarios
    total = (len(range_pix_list) * len(rho0_list) *
             len(anis_mode_list) * len(b_list))
    idx_scn = 0

    for i_r, range_pix in enumerate(range_pix_list):
        for i_rho, rho0 in enumerate(rho0_list):
            for i_am, anis_mode in enumerate(anis_mode_list):
                for i_b, b in enumerate(b_list):
                    idx_scn += 1
                    sc = Scenario(
                        rho0=rho0, range_pix=range_pix, anis_mode=anis_mode,
                        anis_ratio=anis_ratio, angle_deg=angle_deg, nu=nu,
                        n1=n1, n2=n2, B_boot=B_boot, b_block=b, R_mc=R_mc
                    )
                    results = run_mc_study_lmc(
                        sc, H, use_taper=use_taper,
                        tukey_alpha=tukey_alpha,
                        seed=seed_base + (i_r*1000 + i_rho*100 + i_am*10 + i_b)
                    )
                    key = (range_pix, rho0, anis_mode, b)
                    all_out[key] = results
                    label = f"ℓ={range_pix}, ρ0={rho0}, anis={anis_mode}, b={b}"
                    print("\n" + "="*100)
                    print(f"ESCENARIO {idx_scn}/{total}  |  {label}")
                    print("="*100)
                    print_results_table(results, H, header_label=label)
    return all_out

# Gráficas para el Estudio Monte Carlo (BB vs CBB)
# Archivo de entrada: escenarios_codispersion_resumen_126.csv
# Requisitos: solo matplotlib y pandas
# ============================================

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from pathlib import Path

# ---------- Cargar datos ----------
CSV_PATH = "escenarios_codispersion_resumen_126.csv"  # ajusta si está en otra ruta
df = pd.read_csv(CSV_PATH)

# Asegurar tipos
df["Bloque"] = df["Bloque"].astype(int)
df["Rango"] = df["Rango"].astype(int)
df["Rho"] = df["Rho"].astype(float)
df["Método"] = df["Método"].astype(str)
df["Anisotropía"] = df["Anisotropía"].astype(str)

# Orden práctico
orden_bloques = sorted(df["Bloque"].unique())
orden_rangos = sorted(df["Rango"].unique())
orden_rho = sorted(df["Rho"].unique())

# Helper para guardar
OUTDIR = Path("graficas_codispersion")
OUTDIR.mkdir(exist_ok=True)

def savefig(name):
    plt.tight_layout()
    plt.savefig(OUTDIR / f"{name}.png", dpi=300)
    plt.show()

# =========================================================
# 1) Cobertura promedio vs Bloque (líneas), una curva por método y una por rango (en la misma figura)
#    -> agrega bandas de error (±1 sd / sqrt(n)) por bloque y método
# =========================================================
plt.figure(figsize=(10,6))
for r in orden_rangos:
    for metodo in ["BB", "CBB"]:
        d = df[(df["Rango"]==r) & (df["Método"]==metodo)]
        g = d.groupby("Bloque")["Cobertura"]
        x = g.mean().index.values
        y = g.mean().values
        se = g.std(ddof=1).values / np.sqrt(g.count().values)
        label = f"{metodo} | rango={r}"
        plt.plot(x, y, marker="o", label=label)
        plt.fill_between(x, y-se, y+se, alpha=0.15)
plt.axhline(0.95, linestyle="--", linewidth=1)
plt.xlabel("Tamaño de bloque (b)")
plt.ylabel("Cobertura IC 95%")
plt.title("Cobertura vs Bloque por Método y Rango")
plt.legend(ncol=2)
savefig("01_cobertura_vs_bloque_metodo_rango")

# =========================================================
# 2) Cobertura promedio vs Bloque separando por anisotropía (curvas por método)
# =========================================================
plt.figure(figsize=(10,6))
for an in ["iso","aniso"]:
    for metodo in ["BB","CBB"]:
        d = df[(df["Anisotropía"]==an) & (df["Método"]==metodo)]
        g = d.groupby("Bloque")["Cobertura"]
        x = g.mean().index.values
        y = g.mean().values
        se = g.std(ddof=1).values / np.sqrt(g.count().values)
        label = f"{metodo} | {an}"
        plt.plot(x, y, marker="o", label=label)
        plt.fill_between(x, y-se, y+se, alpha=0.15)
plt.axhline(0.95, linestyle="--", linewidth=1)
plt.xlabel("Tamaño de bloque (b)")
plt.ylabel("Cobertura IC 95%")
plt.title("Cobertura vs Bloque por Anisotropía y Método (promedios)")
plt.legend(ncol=2)
savefig("02_cobertura_vs_bloque_anisotropia")

# =========================================================
# 3) Ratio Varianzas (Var* / Var_MC) vs Bloque (curvas por método y rango)
# =========================================================
plt.figure(figsize=(10,6))
for r in orden_rangos:
    for metodo in ["BB","CBB"]:
        d = df[(df["Rango"]==r) & (df["Método"]==metodo)]
        g = d.groupby("Bloque")["Ratio Varianzas"]
        x = g.mean().index.values
        y = g.mean().values
        se = g.std(ddof=1).values / np.sqrt(g.count().values)

plt.axhline(1.0, linestyle="--", linewidth=1)
plt.xlabel("Tamaño de bloque (b)")

plt.legend(ncol=2)
savefig("03_ratio_var_vs_bloque_metodo_rango")

# 4) “Mapa de calor” de cobertura promedio por (Rango × Bloque) — BB y CBB por separado
# (sin subplots: hacemos dos figuras, una por método)
# =========================================================
def heat_matrix(dfm):
    M = []
    for r in orden_rangos:
        fila = []
        for b in orden_bloques:
            dd = dfm[(dfm["Rango"]==r) & (dfm["Bloque"]==b)]["Cobertura"]
            fila.append(dd.mean() if len(dd)>0 else np.nan)
        M.append(fila)
    return np.array(M)

for metodo in ["BB","CBB"]:
    plt.figure(figsize=(10,5))
    M = heat_matrix(df[df["Método"]==metodo])
    im = plt.imshow(M, aspect="auto", interpolation="nearest")
    plt.colorbar(im, label="Cobertura IC 95%")
    plt.xticks(range(len(orden_bloques)), orden_bloques)
    plt.yticks(range(len(orden_rangos)), orden_rangos)
    plt.xlabel("Bloque")
    plt.ylabel("Rango")
    plt.title(f"Cobertura promedio por (Rango×Bloque) — {metodo}")
    savefig(f"04_heat_cobertura_{metodo}")

        for b in orden_bloques:

            fila.append(dd.mean() if len(dd)>0 else np.nan)
        M.append(fila)
    return np.array(M)

for metodo in ["BB","CBB"]:
    plt.figure(figsize=(10,5))
    M = heat_matrix_ratio(df[df["Método"]==metodo])

    plt.title(f"Ratio promedio por (Rango×Bloque) — {metodo}")

labels = []

for rho in orden_rho:

# 9) Curva del “bloque óptimo” por escenario:
#    Para cada (Rho,Rango,Anisotropía,Método), elegir el b con Cobertura más cercana a 0.95
#    y graficar la distribución de esos b* (histograma) por método.
# =========================================================
def bloque_optimo(gr):
    # b con cobertura más cercana a 0.95 (si empata, tomar el menor)
    k = (gr["Cobertura"]-0.95).abs()
    idx = k.idxmin()
    return gr.loc[idx, "Bloque"]

for metodo in ["BB","CBB"]:
    for rho in orden_rho:
        for r in orden_rangos:
            for an in ["iso","aniso"]:
                sub = df[(df["Método"]==metodo) & (df["Rho"]==rho) & (df["Rango"]==r) & (df["Anisotropía"]==an)]
                if len(sub)>0:
                    bstar = bloque_optimo(sub)
                    optimos.append({"Método": metodo, "Bloque*": bstar})
opt_df = pd.DataFrame(optimos)

plt.figure(figsize=(10,6))
bins = np.arange(min(orden_bloques)-1, max(orden_bloques)+2, 4)

plt.ylabel("Densidad")
